Função para calcular a média das notas excluindo a maior e a menor nota

JavaScript 
            
            function calcularMedia(notas) {
              notas = notas.sort(function (a, b) {
                return a - b;
              });

              let notasComputadas = notas.slice(1, 4);
            
              let soma = 0;
              notasComputadas.forEach(function (nota) {
                soma += nota;
              });
            
              return soma / notasComputadas.length;
            }

function calcularMedia(notas) {...}
Esta é a declaração de uma função chamada calcularMedia que recebe um array de notas como argumento.

notas = notas.sort(function (a, b) { return a - b; });
Esta linha ordena o array de notas em ordem crescente.

let notasComputadas = notas.slice(1, 4);
Esta linha cria um novo array notasComputadas que contém todas as notas, exceto a primeira e a última (que são a nota mais baixa e a nota mais alta, respectivamente, devido à ordenação anterior).

let soma = 0;
Esta linha inicializa uma variável soma com o valor 0. Esta variável será usada para somar todas as notas.

notasComputadas.forEach(function (nota) { soma += nota; });
Esta linha percorre cada nota em notasComputadas e a soma à soma.

return soma / notasComputadas.length;
Esta linha calcula a média das notas somadas e retorna o resultado.

Função para apresentar os resultados ao usuário

JavaScript 

            function apresentarResultados(atletas) {
              atletas.forEach(function (atleta) {
                let nome = atleta.nome;
                let notas = atleta.notas;
                let media = calcularMedia(notas);
                let notasObtidas = notas.join(",");
            
                console.log("Atleta: " + nome);
                console.log("Notas Obtidas: " + notasObtidas);
                console.log("Média Válida: " + media.toFixed(2) + "\n");
              });
            }

function apresentarResultados(atletas) {...}
Esta é a declaração de uma função chamada apresentarResultados que recebe um array de atletas como argumento.

atletas.forEach(function (atleta) {...});
Esta linha percorre cada atleta no array atletas.

Dentro da função forEach, são extraídos o nome e as notas de cada atleta, é calculada a media das notas usando a função calcularMedia, e é criada uma string notasObtidas que contém todas as notas separadas por vírgulas.

Em seguida, são impressos no console o nome do atleta, as notas obtidas e a média válida.
